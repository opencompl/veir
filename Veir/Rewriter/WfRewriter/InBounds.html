    InBounds theorems for WfRewriter #

    This file contains lemmas about how the InBounds property of pointers is preserved (or not) by various operations of the WfRewriter. All of these lemmas are annotated by grind, and should in principle never need to be manually invoked.

    @[simp]
    theorem Veir.WfRewriter.setBlockArguments_inBounds_iff {OpInfo : Type} [HasOpInfo OpInfo] {ctx : WfIRContext OpInfo} {blockPtr : BlockPtr} {ptr : GenericPtr} {types : Array TypeAttr} {hblock : blockPtr.InBounds ctx.raw} {noUses : ∀ (blockArg : ValuePtr), blockArg ∈ blockPtr.getArguments! ctx.raw → ¬blockArg.hasUses! ctx.raw = true} :
    ptr.InBounds (setBlockArguments ctx blockPtr types hblock noUses).raw ↔ match ptr with | GenericPtr.blockArgument argPtr => if argPtr.block = blockPtr then argPtr.index < types.size else argPtr.InBounds ctx.raw | GenericPtr.value (ValuePtr.blockArgument argPtr) => if argPtr.block = blockPtr then argPtr.index < types.size else argPtr.InBounds ctx.raw | GenericPtr.opOperandPtr (OpOperandPtrPtr.valueFirstUse (ValuePtr.blockArgument argPtr)) => if argPtr.block = blockPtr then argPtr.index < types.size else argPtr.InBounds ctx.raw | x => ptr.InBounds ctx.raw
